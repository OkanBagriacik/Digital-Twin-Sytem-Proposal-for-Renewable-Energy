import django_filters

from .models import *
from distutils.util import strtobool


class OrderFilter(django_filters.FilterSet):
    class Meta:
        model = Energy
        fields = ['country']
        exclude = ['id']

    def __init__(self, *args):
        super().__init__()
        self.filters['country'].extra['choices'] = args[1] + self.filters['country'].extra['choices']

    country = django_filters.TypedChoiceFilter(choices=(()), coerce=strtobool)

def speResults(request):
    template = loader.get_template('polls/index.html')
    data = Energy.objects.all()
    monthLabels = data.values_list('month', flat=True).distinct()
    monthdict= {}
    for i in range(len(monthLabels)):
        monthdict[i+1] = monthLabels[i]
    list = [(k, v) for k, v in monthdict.items()]
    my_filter = OrderFilter(request.GET, tuple(list))
    context = {"control": "datafilter", "filter": my_filter}
    return HttpResponse(template.render(context, request))
